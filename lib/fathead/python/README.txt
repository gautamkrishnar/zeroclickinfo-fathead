Dependencies:
- Python 3.4.3 or 3.5.2
- pip3 install -r requirements.txt

if using virtualenvwrapper:
- pip3 install virtualenvwrapper
- export VIRTUALENVWRAPPER_PYTHON=/location/of/python3
- source /location/of/virtualenvwrapper.sh
- mkvirtualenv <virtualenvname> --python=/location/of/python3
- pip install -r requirements.txt
